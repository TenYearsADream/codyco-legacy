cmake_minimum_required(VERSION 2.8.11)
SET(PROJECTNAME insituFTSensorCalibration)
PROJECT(${PROJECTNAME})

find_package(YARP REQUIRED)
find_package(ICUB REQUIRED)
find_package(Eigen3 REQUIRED)
find_package(yarpWholeBodyInterface REQUIRED)
find_package(InSituFTCalibration REQUIRED)

#Thrift configuration
include(YarpIDL)

set(generated_libs_dir "${CMAKE_CURRENT_SOURCE_DIR}")
yarp_idl_to_dir(${PROJECTNAME}.thrift ${generated_libs_dir})

set(module_source ./src/module.cpp
                  ./src/main.cpp
                  ./src/insituFTSensorCalibration_IDLServer.cpp)
FILE(GLOB folder_header ./src/*.h)
FILE(GLOB folder_generated_header ${generated_libs_dir}/include/*.h)
set(batch_main_source ./src/main-batch-testing.cpp)
set(shared_source     ./src/ftcalibrationdataset.cpp)

SOURCE_GROUP("Source Files" FILES ${module_source} ${batch_main_source})
SOURCE_GROUP("Header Files" FILES ${folder_header})
SOURCE_GROUP("Autogenerated Header Files" FILES ${folder_generated_header})

#include(iCubHelpers)
#include(iCubOptions)

SET(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} ${ICUB_LINK_FLAGS}")

ADD_EXECUTABLE(${PROJECTNAME} ${module_source} ${folder_header} ${shared_source} ${folder_generated_header})

add_executable(insituFTSensorCalibrationBatch ${shared_source} ${batch_main_source})

INCLUDE_DIRECTORIES(SYSTEM
                    ${YARP_INCLUDE_DIRS}
                    ${EIGEN3_INCLUDE_DIR}
                    ${yarpWholeBodyInterface_INCLUDE_DIRS}
                    ${InSituFTCalibration_INCLUDE_DIRS}
                    ${ctrlLib_INCLUDE_DIRS}
                    ${skinDynLib_INCLUDE_DIRS})

include_directories(${PROJECT_SOURCE_DIR}/src ${generated_libs_dir}/include)

TARGET_LINK_LIBRARIES(${PROJECTNAME} ${YARP_LIBRARIES}
                                     ctrlLib
                                     ${InSituFTCalibration_LIBRARIES}
                                     ${yarpWholeBodyInterface_LIBRARIES})

TARGET_LINK_LIBRARIES(insituFTSensorCalibrationBatch ${YARP_LIBRARIES}
                                     ctrlLib
                                     ${InSituFTCalibration_LIBRARIES}
                                     ${yarpWholeBodyInterface_LIBRARIES})



INSTALL(TARGETS ${PROJECTNAME} DESTINATION bin)
INSTALL(TARGETS insituFTSensorCalibrationBatch DESTINATION bin)

add_subdirectory(app)
add_subdirectory(python)
